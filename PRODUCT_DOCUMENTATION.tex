% Executive Summary
\section{Executive Summary}

\begin{infobox}{CannaCultivate ERP Overview}
\textbf{CannaCultivate ERP} is a comprehensive, compliance-first cultivation management system designed for cannabis cultivation facilities. The system provides end-to-end tracking from seed-to-sale with \textbf{real-time synchronization} across all modules, ensuring regulatory compliance, operational efficiency, and data integrity through a fully integrated Node.js backend architecture with Socket.io for real-time communication.
\end{infobox}

\subsection*{Key Highlights}
\begin{itemize}
    \item[\textcolor{primaryblue}{$\bullet$}] \textbf{Real-Time Integrated System}: Full-stack application with Node.js/Express backend and React frontend, all interconnected via WebSocket (Socket.io) connections
    \item[\textcolor{primaryblue}{$\bullet$}] \textbf{Compliance-First Design}: Built to meet state and federal cannabis regulations with real-time audit trails
    \item[\textcolor{primaryblue}{$\bullet$}] \textbf{Seed-to-Sale Tracking}: Complete traceability with live updates across all modules
    \item[\textcolor{primaryblue}{$\bullet$}] \textbf{Live Monitoring}: Real-time environmental tracking, task management, and inventory control
    \item[\textcolor{primaryblue}{$\bullet$}] \textbf{Scalable Architecture}: Designed to handle multiple facilities with real-time data synchronization
    \item[\textcolor{primaryblue}{$\bullet$}] \textbf{Modern Tech Stack}: React, TypeScript, PostgreSQL, Node.js, Socket.io
\end{itemize}

\subsection*{System Architecture Overview}
\begin{highlightbox}
The system is built as a \textbf{fully integrated real-time application} where:
\begin{itemize}
    \item Frontend (React) communicates with Node.js backend via REST API
    \item All modules are interconnected through WebSocket (Socket.io) for real-time updates
    \item Database changes trigger immediate notifications to all connected clients
    \item Multi-user collaboration with live data synchronization
    \item Real-time dashboard updates, notifications, and alerts
\end{itemize}
\end{highlightbox}

% Project Overview
\section{Project Overview}

\subsection{Business Objectives}
\begin{enumerate}
    \item \textbf{Regulatory Compliance}: Ensure all operations meet state and federal requirements
    \item \textbf{Operational Efficiency}: Streamline cultivation operations and reduce manual errors
    \item \textbf{Data Integrity}: Maintain accurate, auditable records of all operations
    \item \textbf{Scalability}: Support multiple facilities and growing operations
    \item \textbf{User Experience}: Intuitive interface for operators, managers, and administrators
\end{enumerate}

\subsection{Target Users}
\begin{itemize}
    \item \textbf{Cultivation Operators}: Daily plant care and monitoring with real-time task assignments and alerts
    \item \textbf{Facility Managers}: Oversight and resource management with live dashboard updates
    \item \textbf{Compliance Officers}: Real-time audit trails and instant reporting
    \item \textbf{Administrators}: System configuration with live user management and permissions
\end{itemize}

\subsection{System Integration Approach}
\begin{realtimebox}
\textbf{Full-Stack Real-Time Integration:}
\begin{itemize}
    \item \textbf{Frontend}: React application with Socket.io client for real-time communication
    \item \textbf{Backend}: Node.js/Express REST API + Socket.io WebSocket server
    \item \textbf{Database}: PostgreSQL with triggers for real-time event emission
    \item \textbf{Real-Time Layer}: Socket.io for bidirectional communication
    \item \textbf{Event System}: Redis pub/sub for distributed real-time events
    \item All components are \textbf{interconnected} and communicate in real-time
\end{itemize}
\end{realtimebox}

% Features & Functionalities
\section{Features \& Functionalities}

\subsection{Authentication \& Authorization}
\begin{itemize}
    \item \textbf{User Authentication}: Secure login/signup with email and password
    \item \textbf{Role-Based Access Control (RBAC)}: Admin, Manager, Operator, Viewer roles
    \item \textbf{Session Management}: Persistent sessions with secure token handling
    \item \textbf{Multi-Organization Support}: Users can belong to multiple organizations
\end{itemize}

\subsection{Dashboard \& Analytics}
\begin{itemize}
    \item \textbf{Real-time Metrics}: Active plants, batches, rooms, tasks, inventory
    \item \textbf{Capacity Monitoring}: Room utilization and capacity percentages
    \item \textbf{Compliance Score}: Real-time compliance tracking
    \item \textbf{Quick Actions}: Fast access to common operations
    \item \textbf{Visual Charts}: Recharts integration for data visualization
\end{itemize}

\subsection{Cultivation Management}

\subsubsection{Rooms Management}
\begin{itemize}
    \item \textbf{Room Types}: Clone, Vegetative, Flower, Drying rooms
    \item \textbf{Capacity Tracking}: Current vs. maximum plant capacity
    \item \textbf{Environmental Targets}: Temperature, humidity, VPD, light schedules
    \item \textbf{Room Status}: Active/inactive room management
    \item \textbf{Facility Association}: Link rooms to facilities
\end{itemize}

\subsubsection{Genetics Management}
\begin{itemize}
    \item \textbf{Strain Library}: Comprehensive genetics database
    \item \textbf{Strain Information}: THC/CBD potential, flowering time, yield data
    \item \textbf{Lineage Tracking}: Parent strain information
    \item \textbf{Source Tracking}: Clone vs. seed origin
    \item \textbf{Organization-Specific}: Genetics per organization
\end{itemize}

\subsubsection{Mother Plants}
\begin{itemize}
    \item \textbf{Mother Plant Tracking}: Dedicated mother plant management
    \item \textbf{Clone Production}: Track clones produced from mothers
    \item \textbf{Health Monitoring}: Health scores and status tracking
    \item \textbf{Rotation Management}: Mother plant lifecycle management
\end{itemize}

\subsubsection{Plant Batches}
\begin{itemize}
    \item \textbf{Batch Creation}: Create batches with genetics and room assignment
    \item \textbf{Batch Tracking}: Monitor batch progress through growth stages
    \item \textbf{Stage Management}: Clone $\rightarrow$ Vegetative $\rightarrow$ Flowering $\rightarrow$ Harvest
    \item \textbf{Count Tracking}: Initial vs. current plant counts
    \item \textbf{Expected Harvest Dates}: Timeline planning
    \item \textbf{Room Transfers}: Move batches between rooms
\end{itemize}

\subsubsection{Individual Plants}
\begin{itemize}
    \item \textbf{Plant Tagging}: Unique plant identification (METRC-ready)
    \item \textbf{Plant Status}: Active, Quarantine, Harvested, Destroyed
    \item \textbf{Health Scoring}: Numerical health assessment
    \item \textbf{Growth Metrics}: Height, stage, room location
    \item \textbf{Batch Association}: Link plants to batches
    \item \textbf{QR Code Generation}: Physical plant labeling
\end{itemize}

\subsection{Operations Management}

\subsubsection{Environmental Monitoring}
\begin{itemize}
    \item \textbf{Environmental Logs}: Temperature, humidity, CO2, VPD, light intensity
    \item \textbf{Room-Specific Tracking}: Filter logs by room
    \item \textbf{Target vs. Actual}: Compare readings to targets
    \item \textbf{Historical Data}: Trend analysis over time
    \item \textbf{Alert System}: Notifications for out-of-range readings
\end{itemize}

\subsubsection{Feeding Management}
\begin{itemize}
    \item \textbf{Feeding Logs}: Nutrient solution tracking
    \item \textbf{pH \& EC Monitoring}: Solution and runoff measurements
    \item \textbf{Water Volume Tracking}: Gallons per feeding
    \item \textbf{Batch Association}: Link feedings to batches
    \item \textbf{Schedule Management}: Feeding frequency tracking
\end{itemize}

\subsubsection{IPM (Integrated Pest Management)}
\begin{itemize}
    \item \textbf{Treatment Logs}: Pest and disease treatment records
    \item \textbf{Product Tracking}: Chemicals and treatments used
    \item \textbf{PHI/REI Tracking}: Pre-harvest intervals and re-entry intervals
    \item \textbf{Preventive vs. Treatment}: Categorize IPM activities
    \item \textbf{Compliance Tracking}: Ensure regulatory compliance
\end{itemize}

\subsubsection{Task Management}
\begin{itemize}
    \item \textbf{Task Creation}: Create tasks for operations
    \item \textbf{Task Assignment}: Assign to rooms, batches, facilities
    \item \textbf{Priority Levels}: Critical, High, Medium, Low
    \item \textbf{Status Tracking}: Pending, In Progress, Completed
    \item \textbf{Due Dates}: Timeline management
    \item \textbf{Task Templates}: Reusable task definitions
\end{itemize}

\subsection{Harvest \& Inventory}

\subsubsection{Harvest Management}
\begin{itemize}
    \item \textbf{Harvest Batches}: Create harvest records from plant batches
    \item \textbf{Weight Tracking}: Wet weight at harvest
    \item \textbf{Plant Count}: Number of plants harvested
    \item \textbf{Batch Association}: Link to source plant batch
    \item \textbf{Harvest Date Tracking}: Timeline management
\end{itemize}

\subsubsection{Inventory Management}
\begin{itemize}
    \item \textbf{Inventory Lots}: Track finished products
    \item \textbf{Lot Numbers}: Unique lot identification
    \item \textbf{Product Types}: Flower, Trim, Concentrate, Edible, Other
    \item \textbf{Weight Tracking}: Initial and current weights
    \item \textbf{Lab Results}: THC/CBD percentages, moisture content
    \item \textbf{QA Status}: Approved, Pending, Rejected
    \item \textbf{Availability Tracking}: Available for sale status
    \item \textbf{Genetic Association}: Link to source genetics
\end{itemize}

\subsection{Compliance \& Quality}

\subsubsection{Waste Management}
\begin{itemize}
    \item \textbf{Waste Records}: Track all waste disposal
    \item \textbf{Waste Types}: Plant material, trim, failed QA, expired, other
    \item \textbf{Weight Tracking}: Waste weight in pounds
    \item \textbf{Disposal Methods}: Composting, incineration, etc.
    \item \textbf{Reason Tracking}: Why waste was generated
    \item \textbf{Compliance Reporting}: Regulatory waste reporting
\end{itemize}

\subsubsection{Quality Control}
\begin{itemize}
    \item \textbf{QA Testing}: Quality assurance workflows
    \item \textbf{Lab Results}: Integration with testing labs
    \item \textbf{Batch Approval}: Approve/reject inventory lots
    \item \textbf{Compliance Checks}: Regulatory compliance verification
\end{itemize}

\subsubsection{Audit Logs}
\begin{itemize}
    \item \textbf{Complete Audit Trail}: All data changes logged
    \item \textbf{Action Tracking}: INSERT, UPDATE, DELETE operations
    \item \textbf{User Attribution}: Who made changes
    \item \textbf{Timestamp Tracking}: When changes occurred
    \item \textbf{Data Integrity}: Immutable audit records
\end{itemize}

CREATE OR REPLACE FUNCTION notify_plant_change()
RETURNS TRIGGER AS $$
DECLARE
  payload JSON;
  batch_data RECORD;
BEGIN
  SELECT organization_id INTO batch_data
  FROM plant_batches
  WHERE id = COALESCE(NEW.batch_id, OLD.batch_id);
  
  payload = json_build_object(
    'table', 'plants',
    'action', TG_OP,
    'id', COALESCE(NEW.id, OLD.id),
    'organization_id', batch_data.organization_id,
    'batch_id', COALESCE(NEW.batch_id, OLD.batch_id),
    'old_data', row_to_json(OLD),
    'new_data', row_to_json(NEW)
  );
  
  PERFORM pg_notify('plant_changes', payload::text);
  RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER plant_change_trigger
AFTER INSERT OR UPDATE OR DELETE ON plants
FOR EACH ROW EXECUTE FUNCTION notify_plant_change();

// src/socket/socketServer.ts
import { Server } from 'socket.io';
import jwt from 'jsonwebtoken';
import redis from 'redis';

class SocketServer {
  constructor(httpServer) {
    this.io = new Server(httpServer, {
      cors: {
        origin: process.env.FRONTEND_URL,
        methods: ['GET', 'POST'],
        credentials: true
      },
      transports: ['websocket', 'polling']
    });
    
    this.redisClient = redis.createClient();
    this.setupMiddleware();
    this.setupEventHandlers();
    this.setupRedisSubscriber();
  }

  setupMiddleware() {
    // Authentication middleware
    this.io.use(async (socket, next) => {
      try {
        const token = socket.handshake.auth.token;
        const decoded = jwt.verify(token, process.env.JWT_SECRET);
        socket.userId = decoded.userId;
        socket.organizationId = decoded.organizationId;
        next();
      } catch (error) {
        next(new Error('Authentication failed'));
      }
    });
  }

  setupEventHandlers() {
    this.io.on('connection', (socket) => {
      // Join organization room
      socket.join(`org:${socket.organizationId}`);
      
      // Subscribe to specific entities
      socket.on('subscribe:room', (roomId) => {
        socket.join(`room:${roomId}`);
      });
      
      socket.on('subscribe:batch', (batchId) => {
        socket.join(`batch:${batchId}`);
      });
    });
  }

  setupRedisSubscriber() {
    this.redisClient.subscribe('database:changes');
    this.redisClient.on('message', (channel, message) => {
      const event = JSON.parse(message);
      this.broadcastEvent(channel, event);
    });
  }

  broadcastEvent(channel, event) {
    this.io.to(`org:${event.organizationId}`)
      .to(`room:${event.roomId}`)
      .to(`batch:${event.batchId}`)
      .emit('data:update', event);
  }
}
\end{lstlisting}

// src/database/eventListener.ts
import { Client } from 'pg';

class DatabaseEventListener {
  constructor(socketServer) {
    this.socketServer = socketServer;
    this.pgClient = new Client({
      connectionString: process.env.DATABASE_URL
    });
    this.setupListeners();
  }

  async setupListeners() {
    await this.pgClient.connect();
    await this.pgClient.query('LISTEN plant_changes');
    await this.pgClient.query('LISTEN batch_changes');
    
    this.pgClient.on('notification', (msg) => {
      const event = JSON.parse(msg.payload);
      this.handleDatabaseEvent(msg.channel, event);
    });
  }

  handleDatabaseEvent(channel, event) {
    // Publish to Redis
    this.redisClient.publish('database:changes', JSON.stringify(event));
    
    // Emit via Socket.io
    this.socketServer.broadcastEvent('database:changes', event);
    
    // Handle cascading updates
    this.handleCascadingUpdates(channel, event);
  }
}

% Security Architecture
\section{Security Architecture}

\subsection{Authentication \& Authorization}

\subsubsection{Authentication}
\begin{itemize}
    \item \textbf{JWT Tokens}: Access tokens (15 min expiry) + Refresh tokens (7 days)
    \item \textbf{Password Hashing}: bcrypt with salt rounds (12)
    \item \textbf{Rate Limiting}: Login attempts limited (5 per 15 min)
    \item \textbf{2FA}: Optional two-factor authentication
    \item \textbf{Socket.io Authentication}: JWT token required for WebSocket connections
\end{itemize}

\subsubsection{WebSocket Security}
\begin{itemize}
    \item \textbf{Connection Authentication}: JWT token required for Socket.io connection
    \item \textbf{Room-Based Authorization}: Users can only join authorized rooms
    \item \textbf{Event Validation}: All incoming Socket.io events validated
    \item \textbf{Encryption}: WSS (WebSocket Secure) for production
    \item \textbf{Connection Limits}: Max connections per user/IP
\end{itemize}

\subsubsection{Authorization}
\begin{itemize}
    \item \textbf{RBAC}: Role-based access control
    \item \textbf{Permission System}: Fine-grained permissions per role
    \item \textbf{Organization Isolation}: Users can only access their organization's data
    \item \textbf{Row-Level Security}: PostgreSQL RLS for additional security
\end{itemize}

\subsection{Data Security}

\subsubsection{Encryption}
\begin{itemize}
    \item \textbf{At Rest}: Database encryption (PostgreSQL TDE)
    \item \textbf{In Transit}: TLS/SSL for all connections
    \item \textbf{Sensitive Data}: Encrypt PII and financial data
\end{itemize}

\subsubsection{Data Validation}
\begin{itemize}
    \item \textbf{Input Validation}: Server-side validation for all inputs
    \item \textbf{SQL Injection Prevention}: Parameterized queries (ORM)
    \item \textbf{XSS Prevention}: Sanitize user inputs
    \item \textbf{CSRF Protection}: CSRF tokens for state-changing operations
\end{itemize}
